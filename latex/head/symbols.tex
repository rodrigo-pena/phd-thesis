    type=symbol,

}

\newglossaryentry{lp}{%
    type=symbol,
    name={$\| \mathbf{v} \|_p$},
    description={The $\ell_p$ norm $\left ( \sum_{i} |\mathbf{v}_i|^p \right )^{1/p}$ of a vector $\mathbf{v}$, for $p \in (0, \infty)$},
    sort={vector-lp}
}

\newglossaryentry{lp-inf}{%
    type=symbol,
    name={$\| \mathbf{v} \|_\infty$},
    description={The maximum absolute entry in a vector $\mathbf{v}$. Also known as the $\ell_\infty$ norm},
    sort={vector-lp-inf}
}
